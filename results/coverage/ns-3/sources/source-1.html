24 fun alignFile(inputName: String, lineLength: Int, outputName: String) {
25     val writer = File(outputName).bufferedWriter()
26     var currentLineLength = 0
27     fun append(word: String) {
28         if (currentLineLength > 0) {
29             if (word.length + currentLineLength >= lineLength) {
30                 writer.newLine()
31                 currentLineLength = 0
32             } else {
33                 writer.write(" ")
34                 currentLineLength++
35             }
36         }
37         writer.write(word)
38         currentLineLength += word.length
39     }
40     for (line in File(inputName).readLines()) {
41         if (line.isEmpty()) {
42             writer.newLine()
43             if (currentLineLength > 0) {
44                 writer.newLine()
45                 currentLineLength = 0
46             }
47             continue
48         }
49         for (word in line.split(Regex("\\s+"))) {
50             append(word)
51         }
52     }
53     writer.close()
54 }

246         File(inputName).forEachLine { line ->
247             val line2 = line.lowercase()
248             val line3 = line2.toSet().joinToString(separator = "")
249             if (line2 == line3 && line.length > max) {
250                 max = line.length

252             } else if (line2 == line3 && line.length == max) outputLine.append(", $line")
253         }
254         it.write(outputLine.toString())
255     }
