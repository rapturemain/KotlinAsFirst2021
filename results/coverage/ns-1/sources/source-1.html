169 fun averageStockPrice(stockPrices: List<Pair<String, Double>>): Map<String, Double> {
170     val l1 = mutableMapOf<String, Double>()

172     for ((name, cost) in stockPrices) {

180     }

183     }
184     return l1
185 }

309             pairOfIndices = Pair(indexSecond, index)

312         l[number - value] = index

338 fun bagPacking(treasures: Map<String, Pair<Int, Int>>, capacity: Int): Set<String> {
339     val treasuresFiltered = treasures.filterValues { it.first <= capacity }
340     if (treasuresFiltered.isEmpty()) return emptySet()
341 
342     val allWeights = treasuresFiltered.map { it.value.first }
343     val allCosts = treasuresFiltered.map { it.value.second }
344     val totalTreasures = treasuresFiltered.size
345     val bagTable = Array(totalTreasures + 1) { Array(capacity + 1) { 0 } }
346 
347     for (i in 0..totalTreasures) {
348         for (j in 0..capacity) {
349             if (i == 0 || j == 0) bagTable[i][j] = 0
350             else if (allWeights[i - 1] <= j) bagTable[i][j] =
351                 maxOf(allCosts[i - 1] + bagTable[i - 1][j - allWeights[i - 1]], bagTable[i - 1][j])
352             else bagTable[i][j] = bagTable[i - 1][j]
353         }
354     }
355 
356     var totalWeight = bagTable[totalTreasures][capacity]
357     var cost = capacity
358     val bag = mutableSetOf<String>()
359     var bagWeight = 0
360 
361     for (weight in totalTreasures downTo 0) {
362         if (totalWeight <= 0) break
363         if (totalWeight == bagTable[weight - 1][cost]) continue
364         else {
365             val searchPair = (allWeights[weight - 1] to allCosts[weight - 1])
366             for ((key, value) in treasuresFiltered) {
367                 if (value == searchPair) {
368                     bagWeight += value.first
369                     if (bagWeight > capacity) break
370                     bag.add(key)
371                 }
372             }
373             totalWeight -= allCosts[weight - 1]
374             cost -= allWeights[weight - 1]
375         }
376     }
377     return bag
